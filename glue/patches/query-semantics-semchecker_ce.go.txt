//go:build !enterprise

package semantics

func GetSemChecker(stmtType string, txn bool) *SemChecker {
	// n1k1 patch: enable enterprise-level SQL++ semantics (e.g. window
	// functions) in the community build. n1k1 is an interpreter, not a
	// licensed Couchbase Server, and implements these features itself; the
	// enterprise *build tag* is avoided because it pulls cgo deps (eventing-ee).
	return newSemChecker(true, stmtType, txn)
}
